pressStart2P.ts in this directory is a generated base64 embedding of
PressStart2P-Regular.ttf (see that file's own header comment) — regenerate it
with:

  node -e "
    const fs = require('fs');
    const buf = fs.readFileSync('functions/qixxx/_lib/fonts/PressStart2P-Regular.ttf');
    const b64 = buf.toString('base64');
    const lines = [];
    for (let i = 0; i < b64.length; i += 100) lines.push('  \"' + b64.slice(i, i + 100) + '\"');
    fs.writeFileSync('functions/qixxx/_lib/fonts/pressStart2P.ts',
      'export const PRESS_START_2P_TTF_BASE64 =\n' + lines.join(' +\n') + ';\n');
  "
